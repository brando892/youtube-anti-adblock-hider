! =====================================================
! YT Adblock Warning Hider – Future-Proof Version
! =====================================================
!
! Purpose:
! Hides YouTube anti-adblock overlays, even if YouTube
! renames or adds new classes.
!
! Notes:
! - Only hides overlays; video playback and controls unaffected.
! - Fully MV3-compliant.
! - Works on main YouTube, Music, and NoCookie embeds.
!
! Installation (AdGuard recommended method):
! 1) Open AdGuard → Settings → Filters → Custom Filters → Add URL
! 2) Paste raw URL: 
!    https://raw.githubusercontent.com/brando892/youtube-anti-adblock-hider/main/youtube-anti-adblock.txt
! 3) Apply / Save, then reload YouTube.

! =====================================================
! Filter Rules – Future-Proof
! =====================================================

! Catch known overlay classes
youtube.com##.html5-video-player [class*="ytp-ad-player-overlay"]
youtube.com##.html5-video-player [class*="ad-overlay"]
youtube.com##.html5-video-player [class*="overlay"]

! Domain fallbacks
music.youtube.com##.html5-video-player [class*="ytp-ad-player-overlay"]
youtube-nocookie.com##.html5-video-player [class*="ytp-ad-player-overlay"]

! Extra polish — ensure no leftover space
youtube.com##.ytp-ad-player-overlay:style(display: none !important; height: 0 !important;)

! Future-proof: hide any div containing 'ad' or 'overlay' in class inside video player
youtube.com##.html5-video-player [class*="ad"]:style(display: none !important; height: 0 !important;)
youtube.com##.html5-video-player [class*="overlay"]:style(display: none !important; height: 0 !important;)
