! =====================================================
! YT Adblock Warning Hider
! =====================================================
!
! What this does:
! Hides YouTube anti-adblock overlay popups on specified YouTube domains.
! Fully MV3-/MV4-compliant, using cosmetic rules + domain restrictions.
!
! Limitations:
! - Does NOT block video ads.
! - Rare dynamically generated overlays may still appear.
! - CSS-only; cannot override internal YouTube JS logic.

! =====================================================
! Filter Rules
! =====================================================

! (1) Main overlay element (YouTube); domain limited
youtube.com##.html5-video-player [class*="ytp-ad-player-overlay"]$domain=youtube.com

! (2) Alternative overlay class; multiple domains
youtube.com##.html5-video-player [class*="ad-overlay"]$domain=youtube.com|music.youtube.com

! (3) Generic fallback for future class changes
youtube.com##.html5-video-player [class*="overlay"]$domain=youtube.com|youtube-nocookie.com

! (4) Extra polish â€” raise priority, hide leftover blank space
youtube.com##.html5-video-player [class*="ytp-ad-player-overlay"]:style(display:none!important;height:0!important;width:0!important;)$domain=youtube.com|music.youtube.com

! (5) Safe CSS catch-all mimicking JS visual behavior
youtube.com##.html5-video-player [class*="ad"]:style(display:none!important;height:0!important;)$domain=youtube.com|music.youtube.com

! (6) Domain fallbacks
music.youtube.com##.html5-video-player [class*="ytp-ad-player-overlay"]$domain=music.youtube.com
youtube-nocookie.com##.html5-video-player [class*="ytp-ad-player-overlay"]$domain=youtube-nocookie.com
